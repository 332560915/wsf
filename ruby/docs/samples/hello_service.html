def greet(message)
    responsePayloadString = <<XML
            <greetResponse>Hello Client!</greetResponse>
XML;
    return WSMessage.new(responsePayloadString)
end


class HelloServiceController < ApplicationController

  require "wsf"

  def greet
      #operation to ruby function map
      service = WSService.new({"operations" => {"greet" => "greet"}}

      res = service.reply(request, response);

      render :text => res
  end

end
